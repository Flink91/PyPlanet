***OnInit***
***
declare Cb_Public  <=> (Page.GetFirstChild("cb_public") as CMlFrame);
declare Cb_Private <=> (Page.GetFirstChild("cb_private") as CMlFrame);

declare Entry_FolderPrivacy_Value <=> (Page.GetFirstChild("folder_privacy") as CMlEntry);
Entry_FolderPrivacy_Value.Value = "public";

***

***OnMouseClick***
***
if (Event.Control.HasClass("uiCheckboxElement") ) {
    if (Event.Control.Parent.HasClass("uiCheckbox")) {
        if (Event.Control.Parent.ControlId == "cb_public") {
            Entry_FolderPrivacy_Value.Value = "public";
            uiSetCheckbox(Cb_Private, False);
            uiSetCheckbox(Cb_Public, True);
        } else {
            Entry_FolderPrivacy_Value.Value = "private";
            uiSetCheckbox(Cb_Private, True);
            uiSetCheckbox(Cb_Public, False);
        }
    }
}
